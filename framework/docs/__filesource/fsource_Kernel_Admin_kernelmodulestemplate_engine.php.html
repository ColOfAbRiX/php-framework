  159. function build_template( $template, $values = Array(), $obj = null )
    160. 

  160. {
    161. 

  161.     // Attenzione! Questa funzione presenta dei problemi con il caching. In alcune situazioni
    162. 

  162.     // vengono riutilizzati gli stessi parametri impostati per altri template, non ne conosco la
    163. 

  163.     // causa e non so come risolverli. Ho quindi disabilitato il caching.

  164.  
    165. 

  165.     global $SMARTY;
    166. 

  166.  
    167. 

  167.     // Gestione parametri in ingresso
    168. 

  168.     if( !is_a($obj, "Smarty") )
    169. 

  169.     {
    170. 

  170.         $obj = $SMARTY;
    171. 

  171.     }
    172. 

  172.     if( !is_a($obj, "Smarty") )
    173. 

  173.     {
    174. 

  174.         $obj = smarty_create( false );
    175. 

  175.     }
    176. 

  176.  
    177. 

  177.  
    178. 

  178.     // Cancellazione preventiva degli assegnamenti precedenti
    179. 

  179.     $obj->clear_all_assign();
    180. 

  180.  
    181. 

  181.     // Provo ad assegnare le varibili tramite la chiamata di funzioni
    182. 

  182.     $id = debug_timer_start( debug_timer_newid() );
    183. 

  183.     $values = dynamic_assign( $template, $values );
    184. 

  184.  
    185. 

  185.     // Assegno tutti i valori
    186. 

  186.     foreach ($values as $key => $value)
    187. 

  187.     {
    188. 

  188.         $obj->assign( $key, $value );
    189. 

  189.     }
    190. 

  190.  
    191. 

  191.     // Recupero l'output ed elimino i valori inseriti
    192. 

  192.     debug_watch( "Building '$template' with", $values );
    193. 

  193.     
    194. 

  194.     $output = $obj->fetch( $template );
    195. 

  195.     debug_watch( "Building template required", (debug_timer_stop($id) * 1000) . "ms" );
    196. 

  196.     
    197. 

  197.     $obj->clear_all_assign();
    198. 

  198.  
    199. 

  199.  
    200. 

  200.     return $output;
    201. 

  201. }

  203. /**
    204. 

  204.  * Crea un template complesso
    205. 

  205.  *
    206. 

  206.  * Un template complesso è un template che contiene una serie di altri template creati o no in sequenza. Un esempio
    207. 

  207.  * concreto di template complesso può essere una lista di news. Ogni news è descritta da un template, queste news vengono
    208. 

  208.  * poi caricate e quindi sono tutte inserite all'interno di un'altro template.
    209. 

  209.  *
    210. 

  210.  * Descrizione parametro $container:
    211. 

  211.  *   $container["template"]:  Nome del template per il contenitore.
    212. 

  212.  *   $container["list_name"]: Variabile del template contenitore che accoglia la lista di elementi.
    213. 

  213.  *   $container["values"]:    Opzionale. Altri assegnamenti da passare al template.
    214. 

  214.  * Descrizione parametro $items:
    215. 

  215.  *   $items["data"]:          I dati in ingresso sotto forma di un array. La prima dimensione dell'array contiene
    216. 

  216.  *                            la lista dei dati sequenziali e dentro ogni indice c'è un array associativo in cui
    217. 

  217.  *                            ogni chiave è una variabile del template e ogni valore sono i valori delle variabili
    218. 

  218.  *                            nel template.
    219. 

  219.  *   $items["template"]:      Nome del template per un singolo elemento
    220. 

  220.  *   $items["input_filter"]:  Opzionale. Funzione di elaborazione dei dati in ingresso
    221. 

  221.  *   $items["item_function"]: Opzionale. Funzione di elaborazione che viene chiamata per ogni elemento
    222. 

  222.  *   $items["merge_function"]:Opzionale. Funzione personalizzata per l'unione degli elementi
    223. 

  223.  *
    224. 

  224.  * @author     Fabrizio Colonna
    225. 

  225.  * @date       14/12/2005
    226. 

  226.  * @param      Array    $container Dati per il template contenitore dei singoli elementi
    227. 

  227.  * @param      Array    $items Dati per i singoli elementi da elaborare
    228. 

  228.  * @return     String   Il template corretamente formattato

  230. function build_cmplx_tmpl( $container, $items )
    231. 

  231. {
    232. 

  232.     $id = debug_timer_start( debug_timer_newid() );
    233. 

  233.  
    234. 

  234.     $filled = "";
    235. 

  235.     $out_data = Array();
    236. 

  236.  
    237. 

  237.     // Chiamo più facilmente i dati
    238. 

  238.     $in_data = $items["data"];
    239. 

  239.     $input_func = get_index( "input_filter", $items );            // Potrebbe non essere passato
    240. 

  240.     $item_func = get_index( "item_function", $items );            // Potrebbe non essere passato
    241. 

  241.     $custom_merge_func = get_index( "merge_function", $items );    // Potrebbe non essere passato
    242. 

  242.  
    243. 

  243.     // Elaborazione dei dati in ingresso da una funzione personalizzata
    244. 

  244.     if( $input_func )
    245. 

  245.     {
    246. 

  246.         $in_data = $input_func( $in_data );
    247. 

  247.     }
    248. 

  248.  
    249. 

  249.     // Riempio ogni template con un valore della lista
    250. 

  250.     for( $i = 0; $i < count($in_data); $i++ )
    251. 

  251.     {
    252. 

  252.         // Chiamata di funzione per la gestione di un singolo item
    253. 

  253.         if( $item_func )
    254. 

  254.         {
    255. 

  255.             $in_data[$i] = $item_func( $in_data[$i], $i );
    256. 

  256.         }
    257. 

  257.         
    258. 

  258.         if( $custom_merge_func )
    259. 

  259.         {
    260. 

  260.             debug_error( "Sezione da testare" );
    261. 

  261.  
    262. 

  262.             // Chiamata alla funzione di unione personalizzata
    263. 

  263.             $tmp = $custom_merge_func( $in_data, $i );
    264. 

  264.             $out_data = array_merge( $out_data, $tmp );
    265. 

  265.         }
    266. 

  266.         else
    267. 

  267.         {
    268. 

  268.             $out_data[$i] = build_template( $items["template"], $in_data[$i] );
    269. 

  269.         }
    270. 

  270.     }
    271. 

  271.  
    272. 

  272.     // Unisco tutti i valori
    273. 

  273.     $filled = Array( $container["list_name"] => implode("", $out_data) );
    274. 

  274.     if( isset($container["values"]) )
    275. 

  275.     {
    276. 

  276.         $filled = array_merge( $container["values"], $filled );
    277. 

  277.     }
    278. 

  278.  
    279. 

  279.     // Sostituisco la stringa risultante nel template unitario
    280. 

  280.     return build_template( $container["template"], $filled );
    281. 

  281. }

    314.

    321. 

  321.             // Inserisco il risultato assieme alle altre variabili gia assegnate
    322. 

  322.             $values = array_merge( $values, Array($var_name => $var_value) );
    323. 

  323.         }
    324. 

  324.     }

  325.
