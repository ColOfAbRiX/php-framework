  64.     function _get_template( $resource, $type )
    65. 

  65.     {
    66. 

  66.         $out = Array();
    67. 

  67.         $_match = Array();
    68. 

  68.  
    69.

    70. 

  70.         $_params = array( "resource_name" => $resource );
    71. 

  71.         if( !$this->_fetch_resource_info($_params) )
    72. 

  72.         {
    73. 

  73.             return false;
    74. 

  74.         }
    75. 

  75.  
    76.

  77.

  84.  
    85. 

  85.         $this->regexp();
    86. 

  86.  
    87. 

  87.         switch( $type )
    88.

    89. 

  89.             case "variables":
    90. 

  90.                 $reg_exp = $this->_var_regexp;
    91. 

  91.                 break;
    92. 

  92.  
    93. 

  93.             case "functions":
    94. 

  94.                 $reg_exp = $this->_func_regexp;
    95. 

  95.                 break;

  96.         }
    97. 

  97.

  98.         // Trovo le variabili

    102. 

  102.             //$reserved = Array( "literal" );

  103.  
    104. 

  104.             // Comandi riservati di Smarty

  105.             if( $token == "literal" )

  106.             {
    107. 

  107.                 continue;
    108. 

  108.             }
    109. 

  109.  
    110. 

  110.             if( preg_match( "~^$reg_exp$~", $token ) )

  111.             {
    112. 

  112.                 if( !array_search( $token, $out ) )
    113. 

  113.                 {
    114. 

  114.                     array_push( $out, $token );
    115. 

  115.                 }
    116. 

  116.             }
    117. 

  117.         }
    118. 

  118.  
    119.

  122.     function regexp()
    123. 

  123.     {
    124. 

  124.         // matches double quoted strings:
    125. 

  125.         // "foobar"
    126. 

  126.         // "foo\"bar"
    127. 

  127.         $this->_db_qstr_regexp = '"[^"\\\\]*(?:\\\\.[^"\\\\]*)*"';
    128. 

  128.  
    129. 

  129.         // matches single quoted strings:
    130. 

  130.         // 'foobar'
    131. 

  131.         // 'foo\'bar'
    132. 

  132.         $this->_si_qstr_regexp = '\'[^\'\\\\]*(?:\\\\.[^\'\\\\]*)*\'';
    133. 

  133.  
    134. 

  134.         // matches single or double quoted strings
    135. 

  135.         $this->_qstr_regexp = '(?:' . $this->_db_qstr_regexp . '|' . $this->_si_qstr_regexp . ')';
    136. 

  136.  
    137. 

  137.         // matches bracket portion of vars
    138. 

  138.         // [0]
    139. 

  139.         // [foo]
    140. 

  140.         // [$bar]
    141. 

  141.         $this->_var_bracket_regexp = '\[\$?[\w\.]+\]';
    142. 

  142.  
    143. 

  143.         // matches numerical constants
    144. 

  144.         // 30
    145. 

  145.         // -12
    146. 

  146.         // 13.22
    147. 

  147.         $this->_num_const_regexp = '(?:\-?\d+(?:\.\d+)?)';
    148. 

  148.  
    149. 

  149.         // matches $ vars (not objects):
    150. 

  150.         // $foo
    151. 

  151.         // $foo.bar
    152. 

  152.         // $foo.bar.foobar
    153. 

  153.         // $foo[0]
    154. 

  154.         // $foo[$bar]
    155. 

  155.         // $foo[5][blah]
    156. 

  156.         // $foo[5].bar[$foobar][4]
    157. 

  157.         $this->_dvar_math_regexp = '(?:[\+\*\/\%]|(?:-(?!>)))';
    158. 

  158.         $this->_dvar_math_var_regexp = '[\$\w\.\+\-\*\/\%\d\>\[\]]';
    159. 

  159.         $this->_dvar_guts_regexp = '\w+(?:' . $this->_var_bracket_regexp
    160. 

  160.                 . ')*(?:\.\$?\w+(?:' . $this->_var_bracket_regexp . ')*)*(?:' . $this->_dvar_math_regexp . '(?:' . $this->_num_const_regexp . '|' . $this->_dvar_math_var_regexp . ')*)?';
    161. 

  161.         $this->_dvar_regexp = '\$' . $this->_dvar_guts_regexp;
    162. 

  162.  
    163. 

  163.         // matches config vars:
    164. 

  164.         // #foo#
    165. 

  165.         // #foobar123_foo#
    166. 

  166.         $this->_cvar_regexp = '\#\w+\#';
    167. 

  167.  
    168. 

  168.         // matches section vars:
    169. 

  169.         // %foo.bar%
    170. 

  170.         $this->_svar_regexp = '\%\w+\.\w+\%';
    171. 

  171.  
    172. 

  172.         // matches all valid variables (no quotes, no modifiers)
    173. 

  173.         $this->_avar_regexp = '(?:' . $this->_dvar_regexp . '|'
    174. 

  174.            . $this->_cvar_regexp . '|' . $this->_svar_regexp . ')';
    175. 

  175.  
    176. 

  176.         // matches valid variable syntax:
    177. 

  177.         // $foo
    178. 

  178.         // $foo
    179. 

  179.         // #foo#
    180. 

  180.         // #foo#
    181. 

  181.         // "text"
    182. 

  182.         // "text"
    183. 

  183.         $this->_var_regexp = '(?:' . $this->_avar_regexp . '|' . $this->_qstr_regexp . ')';
    184. 

  184.  
    185. 

  185.         // matches valid object call (one level of object nesting allowed in parameters):
    186. 

  186.         // $foo->bar
    187. 

  187.         // $foo->bar()
    188. 

  188.         // $foo->bar("text")

  189.         // $foo->bar($foo, $bar, "text")

    191.

    192.

    193. 

  193.         // $foo->bar($foo->bar)
    194.

    195. 

  195.         // $foo->bar($foo->bar($blah,$foo,44,"foo",$foo[0].bar))
    196. 

  196.         $this->_obj_ext_regexp = '\->(?:\$?' . $this->_dvar_guts_regexp . ')';
    197. 

  197.         $this->_obj_restricted_param_regexp = '(?:'
    198. 

  198.                 . '(?:' . $this->_var_regexp . '|' . $this->_num_const_regexp . ')(?:' . $this->_obj_ext_regexp . '(?:\((?:(?:' . $this->_var_regexp . '|' . $this->_num_const_regexp . ')'
    199. 

  199.                 . '(?:\s*,\s*(?:' . $this->_var_regexp . '|' . $this->_num_const_regexp . '))*)?\))?)*)';
    200. 

  200.         $this->_obj_single_param_regexp = '(?:\w+|' . $this->_obj_restricted_param_regexp . '(?:\s*,\s*(?:(?:\w+|'
    201. 

  201.                 . $this->_var_regexp . $this->_obj_restricted_param_regexp . ')))*)';
    202. 

  202.         $this->_obj_params_regexp = '\((?:' . $this->_obj_single_param_regexp

  203.                 . '(?:\s*,\s*' . $this->_obj_single_param_regexp . ')*)?\)';
    204. 

  204.         $this->_obj_start_regexp = '(?:' . $this->_dvar_regexp . '(?:' . $this->_obj_ext_regexp . ')+)';
    205. 

  205.         $this->_obj_call_regexp = '(?:' . $this->_obj_start_regexp . '(?:' . $this->_obj_params_regexp . ')?(?:' . $this->_dvar_math_regexp . '(?:' . $this->_num_const_regexp . '|' . $this->_dvar_math_var_regexp . ')*)?)';
    206. 

  206.  
    207. 

  207.         // matches valid modifier syntax:

  208.         // |foo
    209. 

  209.         // |@foo
    210. 

  210.         // |foo:"bar"
    211. 

  211.         // |foo:$bar

  214.         // |foo:$foo->bar

  215.         $this->_mod_regexp = '(?:\|@?\w+(?::(?:\w+|' . $this->_num_const_regexp . '|'
    216. 

  216.            . $this->_obj_call_regexp . '|' . $this->_avar_regexp . '|' . $this->_qstr_regexp .'))*)';
    217. 

  217.  
    218. 

  218.         // matches valid function name:

  219.         // foo123
    220. 

  220.         // _foo_bar
    221. 

  221.         $this->_func_regexp = '[a-zA-Z_]\w*';
    222. 

  222.  
    223. 

  223.         // matches valid registered object:
    224. 

  224.         // foo->bar
    225. 

  225.         $this->_reg_obj_regexp = '[a-zA-Z_]\w*->[a-zA-Z_]\w*';
    226. 

  226.  
    227. 

  227.         // matches valid parameter values:
    228. 

  228.         // true
    229. 

  229.         // $foo
    230. 

  230.         // $foo|bar
    231. 

  231.         // #foo#
    232. 

  232.         // #foo#|bar
    233. 

  233.         // "text"
    234. 

  234.         // "text"|bar
    235. 

  235.         // $foo->bar
    236. 

  236.         $this->_param_regexp = '(?:\s*(?:' . $this->_obj_call_regexp . '|'
    237. 

  237.            . $this->_var_regexp . '|' . $this->_num_const_regexp  . '|\w+)(?>' . $this->_mod_regexp . '*)\s*)';
    238. 

  238.  
    239. 

  239.         // matches valid parenthesised function parameters:
    240. 

  240.         //
    241. 

  241.         // "text"
    242. 

  242.         //    $foo, $bar, "text"
    243. 

  243.         // $foo|bar, "foo"|bar, $foo->bar($foo)|bar
    244. 

  244.         $this->_parenth_param_regexp = '(?:\((?:\w+|'
    245. 

  245.                 . $this->_param_regexp . '(?:\s*,\s*(?:(?:\w+|'
    246. 

  246.                 . $this->_param_regexp . ')))*)?\))';
    247. 

  247.  
    248. 

  248.         // matches valid function call:
    249. 

  249.         // foo()
    250. 

  250.         // foo_bar($foo)
    251. 

  251.         // _foo_bar($foo,"bar")
    252. 

  252.         // foo123($foo,$foo->bar(),"foo")
    253. 

  253.         $this->_func_call_regexp = '(?:' . $this->_func_regexp . '\s*(?:'
    254. 

  254.            . $this->_parenth_param_regexp . '))';
    255. 

  255.     }
