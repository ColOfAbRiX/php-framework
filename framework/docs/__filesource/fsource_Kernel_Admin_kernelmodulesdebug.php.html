  87. /**
    88. 

  88.  * Filtra un dato
    89. 

  89.  * 
    90. 

  90.  * Il filtro si intende eseguito su ognuna delle singore righe del debug. E' possibile esprimere
    91. 

  91.  * espressioni booleane, anche complesse, sui dati che compaiono nella visualizzazione del debug per
    92. 

  92.  * ottenere un sottoinsieme ristretto di informazioni, generalmente più utili, delle informazioni
    93. 

  93.  * complete.<br>
    94. 

  94.  * Operatori di confronto: A<B, A<=B, A==B, A!=B, A>=B, A>B, A has "text", A regexp /regexp/imsxeuADSUX<br>
    95. 

  95.  * Operatori logici: (expr1)and(expr2), (expr1)or(expr2), (expr1)xor(expr2), not(expr)<br>
    96. 

  96.  * Variabili di controllo: variabile $data<br>
    97. 

  97.  * Esempio: <samp>(type_data.value.items regexp /<[^<>]>/m) or (type_data.value.item regexp /<[^<>]>/m)</samp>
    98. 

  98.  *
    99. 

  99.  * @author     Fabrizio Colonna
    100. 

  100.  * @date       15/09/2007
    101. 

  101.  * @see        debug_show
    102. 

  102.  * @param      String   $filter La stringa filtro
    103. 

  103.  * @param      Array    $data Dati dell'item
    104. 

  104.  * @return     Bool     True se il dato è valido, false altrimenti
    105. 

  105.  */

  106. function debug_filter_check( $filter, $data )
    107. 

  107. {
    108. 

  108.     // Mettere un controllo sul formato degli operatori logici
    109. 

  109.     //$filter = "(type_data.value.items regexp /<[^<>]>/m) or (type_data.value.item regexp /<[^<>]>/m)";
    110. 

  110.     
    111. 

  111.     // Il filtro vuoto fa passare tutto
    112. 

  112.     if( $filter == "" )
    113. 

  113.     {
    114. 

  114.         return true;
    115. 

  115.     }
    116. 

  116.  
    117. 

  117.     // RegEx di controllo
    118. 

  118.     $l_value     =  "[\w\d\.]*";
    119. 

  119.     $number      =  "\d+(?:\.\d*|)";
    120. 

  120.     $string      =  "\"[^\"]*\"";
    121. 

  121.     $regexp      =  "\/[^\/]*\/(?:[msxeuADSUX]{0,1}){0,10}";
    122. 

  122.     $r_value     =  "(?:$number|$string|$regexp)";
    123. 

  123.     $compare_ops =  "(?:<=|==|!=|>=|<|>|\shas\s|\sregexp\s)";
    124. 

  124.     $pattern     =  "/\s*($l_value\s*$compare_ops\s*$r_value)\s*/i";
    125. 

  125.     // /\s*([\w\d\.]*\s*(?:<=|==|!=|>=|<|>|\shas\s|\sregexp\s)\s*(?:\d+(?:\.\d*|)|"[^"]*"|\/[^\/]*\/(?:[msxeuADSUX]{0,1}){0,10}))\s*/i
    126. 

  126.     
    127. 

  127.     preg_match_all( $pattern, $filter, $output );
    128. 

  128.  
    129. 

  129.     // Sostituisco ogni singola espressione con il suo risultato booleano
    130. 

  130.     foreach( $output[0] as $expression )
    131. 

  131.     {
    132. 

  132.         $original_expr = $expression;
    133. 

  133.         
    134. 

  134.         // Recupero la parte a dx e a sx dell'operatore per eseguire delle operazioni
    135. 

  135.         list( $right, $left ) = preg_split( "/\s*$compare_ops\s*/", $expression, 2 );
    136. 

  136.         
    137. 

  137.         // Elabotazione di R-Value
    138. 

  138.         $right_new = "\$data";        // -> Accesso all'array $data
    139. 

  139.         $array_indexes = split( "\.", $right );
    140. 

  140.         
    141. 

  141.         foreach( $array_indexes as $index )
    142. 

  142.         {
    143. 

  143.             if( !is_numeric($index) )
    144. 

  144.             {
    145. 

  145.                 $right_new .= "[\"$index\"]";
    146. 

  146.             }    
    147. 

  147.             else
    148. 

  148.             {
    149. 

  149.                 $right_new .= "[$index]";
    150. 

  150.             }
    151. 

  151.         }
    152. 

  152.         
    153. 

  153.         // Gli operatori determinano il formato della stringa da eseguire
    154. 

  154.         $operator = str_replace( $right, "", $expression );
    155. 

  155.         $operator = str_replace( $left, "", $operator );
    156. 

  156.         $operator = preg_replace( "/\s*/", "", $operator );
    157. 

  157.         
    158. 

  158.         switch( strtolower($operator) )
    159. 

  159.         {
    160. 

  160.             case "has":        // Operatore "contiene"
    161. 

  161.                 $expression = "strlen(stristr( $right_new, $left )) > 0";
    162. 

  162.                 break;
    163. 

  163.     
    164. 

  164.             case "regexp":    // Regular Expression
    165. 

  165.                 $expression = "preg_match( \"$left\", $right_new ) > 0";
    166. 

  166.                 break;
    167. 

  167.     
    168. 

  168.             default:        // Classici operatori di confronto
    169. 

  169.                 $expression = str_replace( $right, $right_new, $expression );
    170. 

  170.                 $expression = str_replace( $left, $left, $expression );
    171. 

  171.                 $expression = str_replace( $operator, $operator, $expression );
    172. 

  172.                 break;
    173. 

  173.         }
    174. 

  174.     
    175. 

  175.         // Valutazione dell'output
    176. 

  176.         $expr_result = eval( "return ($expression);" ) ? "true" : "false";
    177. 

  177.  
    178. 

  178.         
    179. 

  179.         // Inserimento nella query filtro
    180. 

  180.         $filter = str_replace( $original_expr, $expr_result, $filter );
    181. 

  181.     }
    182. 

  182.  
    183. 

  183.     $filter = preg_replace( "/\s+/", " ", $filter );        // Causa errore con gli spazi
    184. 

  184.     $output = eval( "return ($filter);" );
    185. 

  185.     
    186. 

  186.     return $output;
    187. 

  187. }

  205. function debug_create_item( $type, $type_data, $stack_skip, $extra_skip )
    206. 

  206. {
    207. 

  207.     global $debug_sections;
    208. 

  208.     
    209. 

  209.     // Recupero la sezione di codice specificata per ultima
    210. 

  210.     $current_section = "";
    211. 

  211.     if( !empty($debug_sections) )
    212. 

  212.     {
    213. 

  213.         $current_section = $debug_sections[count($debug_sections) - 1];
    214. 

  214.     }
    215. 

  215.     
    216. 

  216.     // Elimino lo stack trace 
    217. 

  217.     $stack_trace = debug_backtrace();
    218. 

  218.     for( ; $extra_skip > 0; $extra_skip-- )
    219. 

  219.     {
    220. 

  220.         if( isset($stack_trace) )
    221. 

  221.         {
    222. 

  222.             array_shift( $stack_trace );
    223. 

  223.         }
    224. 

  224.     }
    225. 

  225.  
    226. 

  226.     $item_info = Array(
    227. 

  227.         "type" => $type,
    228. 

  228.         "type_data" => $type_data,
    229. 

  229.         "stack_trace" => $stack_trace,
    230. 

  230.         "skip_stack" => $stack_skip,
    231. 

  231.         "time" => microtime( true ),
    232. 

  232.         "tags" => $current_section
    233. 

  233.     );
    234. 

  234.     
    235. 

  235.     // Aggiungo il marker alla lista di debug
    236. 

  236.     debug_add_item( $item_info );
    237. 

  237. }
