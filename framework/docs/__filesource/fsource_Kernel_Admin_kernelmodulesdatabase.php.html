  190. function db_execute( $query, $parameters = Array(), $db = null )
    191. 

  191. {
    192. 

  192.     global $DATABASE, $CONFIG;
    193. 

  193.  
    194. 

  194.     $volatile = (bool)$CONFIG["db"]["selective"];
    195. 

  195.  
    196. 

  196.     // Gestione parametri in ingresso
    197. 

  197.     if( !is_object($db) )
    198. 

  198.     {
    199. 

  199.         $db = $DATABASE;
    200. 

  200.     }
    201. 

  201.     if( !is_object($db) )
    202. 

  202.     {
    203. 

  203.         $db = db_connect( !$volatile, true );        // Si toglie l'impostazione di connessione selettiva
    204. 

  204.     }
    205. 

  205.  
    206. 

  206.     // NOTA: L'escaping dei parametri di ingresso per aumentare la sicurezza non è necessario perchè se ne occupa la
    207. 

  207.     // funzione prepare
    208. 

  208.  
    209. 

  209.     // Preparo ed eseguo la query sostituendo i parametri
    210. 

  210.     $sth = $db->prepare( $query );
    211. 

  211.     $out = $db->execute( $sth, $parameters );
    212. 

  212.  
    213. 

  213.     if( $CONFIG["debug"] )
    214. 

  214.     {
    215. 

  215.         debug_watch( "Executed Query", $db->last_query );
    216. 

  216.     }
    217. 

  217.  
    218. 

  218.     pear_check( $out );
    219. 

  219.  
    220. 

  220.     // Connessione non persistente
    221. 

  221.     if( $volatile )
    222. 

  222.     {
    223. 

  223.         db_disconnect( $db );
    224. 

  224.     }
    225. 

  225.  
    226. 

  226.     return $out;
    227. 

  227. }
