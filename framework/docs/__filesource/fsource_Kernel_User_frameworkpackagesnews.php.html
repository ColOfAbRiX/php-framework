  12.  * <b>Note sul formato del pattern</b> Il pattern di ripetizione delle news è
    13. 

  13.  * composto da più elementi separati tra loro da un punto e virgola. Ogni elemento
    14. 

  14.  * ha il seguente formato: <start>,<type>[/num],<end>

  15.  * - <start>: La news di partenza
    16. 

  16.  * - <type> : Il tipo di visualizzazione (full, reduced, list)

    18. 

  18.  * - <end>  : La news finale

    20. 

  20.  * presente <end> l'algoritmo va avanti fino alla fine delle news. Sono possibili delle
    21. 

  21.  * sovrapposizioni tra i vari <start> ed <end> dei vari elementi, l'ultimo elemento
    22. 

  22.  * della stringa ha la priorità. Ecco un esempio di pattern:
    23. 

  23.  *   "0,full,5;-,list,10;6,reduced/2,10;-,list,-".
    24. 

  24.  * Il significato di questo pattern è: elementi da 1 a 5 full, elementi da 6 a 10 reduced,
    25. 

  25.  * i restanti list. Gli elementi list dal 6 al 10 sono rimpiazzati da quelli reduced. Un
    26.

  57. function compile_pattern( $text_pattern )
    58. 

  58. {
    59. 

  59.     $sections = explode( ";", $text_pattern );
    60. 

  60.     $pattern = Array();
    61. 

  61.     $last_start = 0;
    62. 

  62.  
    63. 

  63.     for( $i = 0; $i < count($sections); $i++ )
    64. 

  64.     {
    65. 

  65.         list( $start, $tmpl, $end ) = explode( ",", $sections[$i] );
    66. 

  66.  
    67. 

  67.         // Controllo gli indici particolari
    68. 

  68.         if( !is_numeric($start) ) {
    69. 

  69.             $start = $last_start;
    70. 

  70.         }
    71. 

  71.         if( !is_numeric($end) ) {
    72. 

  72.             $end = $start + 1;
    73. 

  73.         }
    74. 

  74.  
    75. 

  75.         for( $j = (int)$start; $j < (int)$end; $j++ ) {
    76. 

  76.             $pattern[$j] = $tmpl;
    77. 

  77.         }
    78. 

  78.  
    79. 

  79.         // Memorizzo l'ultimo indice di partenza
    80. 

  80.         $last_start = $end;
    81. 

  81.     }
    82. 

  82.  
    83. 

  83.     return $pattern;
    84. 

  84. }

    113.

  114.

  115.     for( $i = -1; $i < count($news); $i++ )

  227. function get_all_news( $pattern )
    228. 

  228. {
    229. 

  229.     global $CONFIG, $SQL;
    230. 

  230.     $last_pattern = "full/1";
    231. 

  231.     $output = $create = Array();
    232. 

  232.     $i = $iType = 0;
    233. 

  233.     $i_type = $i_news = $last_news = 0;
    234. 

  234.  
    235. 

  235.     // Recupero le news dal database
    236. 

  236.     $news = db_execute_array( $SQL["news_all"], Array($CONFIG["news"]["max"]) );
    237. 

  237.  
    238. 

  238.     // Nota: il formato di $news e di $pattern non possono essere uguali perchè il pattern
    239. 

  239.     // può avere dei buchi e delle sovrapposizioni, mentre le news sono per forza continue
    240. 

  240.  
    241. 

  241.     for( $i = 0; $i < count($news); $i++, $i_news++ )
    242. 

  242.     {
    243. 

  243.         if( empty($news[$i]) ) { continue; }        // Per via di errori nella lettura del database
    244. 

  244.  
    245. 

  245.         // Il pattern può non cominciare da zero oppure può avere dei buchi
    246. 

  246.         if( empty($pattern[$i]) )
    247. 

  247.         {
    248. 

  248.             $pattern[$i] = $last_pattern;
    249. 

  249.         }
    250. 

  250.  
    251. 

  251.         // Cambiamento di pattern, quindi un nuovo indice dei tipi e reset di quello delle news
    252. 

  252.         if( $pattern[$i] != $last_pattern && $i != 0 )
    253. 

  253.         {
    254. 

  254.             if( $last_news >= 1 )
    255. 

  255.             {
    256. 

  256.                 // Il numero delle news non è divisibile per il numero di news per template
    257. 

  257.                 $output[$i_type][$i_news] = get_news_box( $create, $tmpl );
    258. 

  258.                 $create = Array();
    259. 

  259.                 $last_news = 0;
    260. 

  260.             }
    261. 

  261.             $i_type++;
    262. 

  262.             $i_news = 0;
    263. 

  263.         }
    264. 

  264.  
    265. 

  265.         // Creazione di un nuovo tipo nell'array di output
    266. 

  266.         if( count($output) - 1 < $i_type ) {
    267. 

  267.             $output[$i_type] = Array();
    268. 

  268.         }
    269. 

  269.  
    270. 

  270.         // Recupero template e news per template
    271. 

  271.         if( $last_news == 0 )
    272. 

  272.         {
    273. 

  273.             list( $tmpl, $last_news ) = explode( "/", $pattern[$i] . "/" );
    274. 

  274.             if( empty($last_news) ) $last_news = "1";
    275. 

  275.             $last_news = (int)$last_news;
    276. 

  276.         }
    277. 

  277.  
    278. 

  278.         // Tengo memoria di tutte le news
    279. 

  279.         array_push( $create, $news[$i] );
    280. 

  280.  
    281. 

  281.         if( $last_news == 1 )
    282. 

  282.         {
    283. 

  283.             // Posso visualizzare le news
    284. 

  284.             $output[$i_type][$i_news] = get_news_box( $create, $tmpl );
    285. 

  285.             $create = Array();
    286. 

  286.         }
    287. 

  287.         else
    288. 

  288.         {
    289. 

  289.             $i_news--;
    290. 

  290.         }
    291. 

  291.  
    292. 

  292.         $last_news--;
    293. 

  293.         $last_pattern = $pattern[$i];
    294. 

  294.     }
    295. 

  295.  
    296. 

  296.     return $output;
    297.
