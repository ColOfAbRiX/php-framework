  102. function buffer_add_level( $initial_value = "", $buffer_handler = "", $handler_file = "", $level = -1 )
    103. 

  103. {
    104. 

  104.     global $buffer_handlers;
    105. 

  105.  
    106. 

  106.     //se l'utente introduce un livello troppo basso, lo metto come primo livello utente
    107. 

  107.     if( ($level < 1) && ($level != -1) ) $level = 1;
    108. 

  108.  
    109. 

  109.     $last_level = ob_get_level();
    110. 

  110.     $level_difference = $last_level - $level;
    111. 

  111.  
    112. 

  112.     // se il livello è pari a -1 (valore di default) oppure oltre l'ultimo creato,
    113. 

  113.     // il livello del nuovo buffer deve essere impostato al livello inferiore
    114. 

  114.     // rispetto all'ultimo
    115. 

  115.     if( ($level >= $last_level) || ($level == -1) )
    116. 

  116.     {
    117. 

  117.         ob_start();
    118. 

  118.         $buffer_handlers[$last_level + 1] = Array( "priorities" => Array() );
    119. 

  119.         buffer_add_handler( $buffer_handler, $handler_file, $last_level + 1 );
    120. 

  120.         echo( $initial_value );
    121. 

  121.     }
    122. 

  122.     // altrimenti dobbiamo andare a inserire il nostro nuovo buffer prima di quelli che lo
    123. 

  123.     // devono seguire
    124. 

  124.     else
    125. 

  125.     {
    126. 

  126.         $tmp_buf = Array();
    127. 

  127.         for( $i = 0; $i < $level_difference; $i++)
    128. 

  128.         {
    129. 

  129.             // inserisco nell'ultimo elemento dell'array il contenuto del livello a priorità
    130. 

  130.             // maggiore tra i buffer e lo elimino
    131. 

  131.             $tmp_buf[$level_difference - $i] = ob_get_clean();
    132. 

  132.         }
    133. 

  133.  
    134. 

  134.         // prima di ricostruire la struttura dei buffer devo inserire il nuovo gestore
    135. 

  135.         // nell'array degli handlers, perciò spezzo l'array in due parti (la prima con $level elementi,
    136. 

  136.         // la seconda con la differenza tra il numero di elementi di $buffer_handlers e $level elementi),
    137. 

  137.         // $first_half e $second_half e ci inserisco in mezzo il nuovo gestore, quindi faccio il
    138. 

  138.         // merge delle due parti per ricomporre l'array $buffer_handlers
    139. 

  139.  
    140. 

  140.         $first_half = array_slice( $buffer_handlers, 0, $level );
    141. 

  141.         $second_half = array_slice( $buffer_handlers, $level, count($buffer_handlers) );
    142. 

  142.  
    143. 

  143.         //array_push( $first_half, $buffer_handler );
    144. 

  144.         array_push( $first_half, Array() );
    145. 

  145.  
    146. 

  146.         // Ora l'array di array buffer_handlers può essere ricomposto
    147. 

  147.         $buffer_handlers = array_merge( $first_half, $second_half );
    148. 

  148.  
    149. 

  149.         // a questo punto abbiamo in mano un array che contiene tutti i dati dei buffer ai livelli
    150. 

  150.         // che hanno indice inferiore rispetto a quello che dobbiamo inserire
    151. 

  151.         buffer_add_handler( $buffer_handler, $handler_file, $level );
    152. 

  152.         ob_start();
    153. 

  153.         echo( $initial_value );
    154. 

  154.  
    155. 

  155.         // a questo punto possiamo reinserire i buffer che devono adesso seguire il nuovo
    156. 

  156.         for( $i = 0; $i < $level_difference; $i++ )
    157. 

  157.         {
    158. 

  158.             // ricreo il buffer che viene gestito dal nostro sistema e non da php
    159. 

  159.             ob_start();
    160. 

  160.             // valorizzazione del buffer
    161. 

  161.             echo( $tmp_buf[$i] );
    162. 

  162.         }
    163. 

  163.     }
    164. 

  164. }

  216. function buffer_add_handler( $handler, $handler_file, $level = -1 )
    217. 

  217. {
    218. 

  218.     global $buffer_handlers;
    219. 

  219.  
    220. 

  220.     // Controllo della validità dei dati in ingresso
    221. 

  221.     if( $level < 1 && $level != -1 )
    222. 

  222.     {
    223. 

  223.         $level = 1;
    224. 

  224.     }
    225. 

  225.     elseif( $level > count($buffer_handlers) || $level == -1  )
    226. 

  226.     {
    227. 

  227.         $level = count($buffer_handlers) - 1;
    228. 

  228.     }
    229. 

  229.     if( $handler == "" || $handler_file == "" )
    230. 

  230.     {
    231. 

  231.         return;
    232. 

  232.     }
    233. 

  233.  
    234. 

  234.     // Aggiungo il nuovo gestore. Il controllo della sua esistenza viene fatto durante la chiamata
    235. 

  235.     // Per memorizzare il file in cui si trova la funzione il nome della funzione viene utilizzato come chiave dell'array
    236. 

  236.     // e il nome del file come valore. La priorità dei gestori viene memorizzata in un indice a parte.
    237. 

  237.     $level_handlers = $buffer_handlers[$level];
    238. 

  238.  
    239. 

  239.     // Array dei gestori
    240. 

  240.     $level_handlers = array_merge( $level_handlers, Array($handler => $handler_file) );
    241. 

  241.  
    242. 

  242.     // Array delle priorità
    243. 

  243.     $priorities = $level_handlers["priorities"];
    244. 

  244.     array_init( $priorities );
    245. 

  245.     if( !in_array($handler, $priorities) )
    246. 

  246.     {
    247. 

  247.         array_unshift( $priorities, $handler );
    248. 

  248.     }
    249. 

  249.     $level_handlers["priorities"] = $priorities;
    250. 

  250.  
    251. 

  251.     $buffer_handlers[$level] = $level_handlers;
    252. 

  252. }

  269. function buffer_remove_handler( $level, $handler, $remove_all = false )
    270. 

  270. {
    271. 

  271.     global $buffer_handlers;
    272. 

  272.  
    273. 

  273.     // Controllo della validità dei dati in ingresso
    274. 

  274.     if( $level < 1 )
    275. 

  275.     {
    276. 

  276.         $level = 1;
    277. 

  277.     }
    278. 

  278.     elseif( $level > count($buffer_handlers) )
    279. 

  279.     {
    280. 

  280.         $level = count($buffer_handlers);
    281. 

  281.     }
    282. 

  282.     if( $handler == "" )
    283. 

  283.     {
    284. 

  284.         return;
    285. 

  285.     }
    286. 

  286.  
    287. 

  287.     // Recupero i gestori per il livello
    288. 

  288.     $level_handlers = $buffer_handlers[$level];
    289. 

  289.  
    290. 

  290.     if( $remove_all )
    291. 

  291.     {
    292. 

  292.         // Elimino tutti gli handler
    293. 

  293.         $level_handlers = Array();
    294. 

  294.         $level_handlers["priorities"] = Array();
    295. 

  295.     }
    296. 

  296.     else
    297. 

  297.     {
    298. 

  298.         // Elimino l'handler
    299. 

  299.         $level_handlers = array_diff_key( $level_handlers, Array($handler => null) );
    300. 

  300.         $level_handlers["priorities"] = array_diff( $level_handlers["priorities"], Array($handler) );
    301. 

  301.     }
    302. 

  302.  
    303. 

  303.     $buffer_handlers[$level] = $level_handlers;
    304. 

  304. }

    319. 

  319. {
    320. 

  320.     global $buffer_handlers;
    321. 

  321.  
    322. 

  322.     $level = ob_get_level();
    323. 

  323.  
    324. 

  324.     $level_handlers = $buffer_handlers[$level];

  325.     $priorities = $level_handlers["priorities"];

  326.

  334.         $file = dirname(__FILE__)."/../$file";

  336.         call_ext_function( $file, $handler, Array($buffer), $buffer );
